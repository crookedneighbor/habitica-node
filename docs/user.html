get()

Gets the entire user object.

user.get()
  .then((user) => {
    user._id; // your user id
    user.todso; // an array of your todos
    user.items; // your items object
  });

  get () {
    return this._connection.get('user')
      .then((user) => {
        return user;
      })
      .catch((err) => {
        throw err;
      });
  }
